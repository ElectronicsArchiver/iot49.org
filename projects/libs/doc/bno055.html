%softreset

import time
from machine import I2C as I2C
from bno055 import BNO055

i2c = I2C(1, freq=400_000)

imu = BNO055(i2c)

# otherwise first result is all zeros ...
time.sleep(0.1)

for i in range(5):
    print("acceleration", imu.accel())
    print("rate        ", imu.gyro())
    print("euler       ", imu.euler())
    print("quaternion  ", imu.quaternion())
    time.sleep(1)
    print()
